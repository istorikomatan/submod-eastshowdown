# РФ принудительно покидает Киев
RUS_leave_kiev = {
	allowed = {
		original_tag = RUS
	}

	enable = {
		AND = {
			is_historical_focus_on = yes
			has_war_with = UKR
			date > 2022.3.20
		}
	}

	abort = {
		OR = {
			date > 2022.4.1
			NOT = {
				has_war_with = UKR
			}
		}
	}

	ai_strategy = {
		type = front_unit_request
		strategic_region = 23
		strategic_region = 45
		value = -10000
	}
}
